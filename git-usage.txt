#配置用户名
git config --global user.name username

#配置邮箱
git config --global user.email aa@email.com

#项目初始化
git init

#放文件到仓库
git add filename

#查看状态
git status

#文件上架
git commit -m "first commit"

#回退
git reset --hard HEAD^
git reset --hard HEAD^^
git reset --hard HEAD~10
git reset --hard 具体版本号

#命令历史
git reflog

#回退到最后一次add 或者commit
git checkout -- git-usage.txt
